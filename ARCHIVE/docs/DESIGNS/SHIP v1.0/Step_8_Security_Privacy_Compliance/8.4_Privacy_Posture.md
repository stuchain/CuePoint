**8.4.2.3 Data Retention Policies**

    ```python
    # SRC/cuepoint/utils/privacy.py
    from pathlib import Path
    from datetime import datetime, timedelta
    import time
    
    class DataRetentionManager:
        """Manage data retention policies."""
        
        @staticmethod
        def enforce_cache_retention(cache_dir: Path, max_size_mb: int = 100) -> None:
            """Enforce cache size retention.
            
            Args:
                cache_dir: Cache directory
                max_size_mb: Maximum cache size in MB
            """
            max_size_bytes = max_size_mb * 1024 * 1024
            
            # Calculate current cache size
            total_size = sum(f.stat().st_size for f in cache_dir.rglob('*') if f.is_file())
            
            if total_size > max_size_bytes:
                # Delete oldest files first
                files = [(f, f.stat().st_mtime) for f in cache_dir.rglob('*') if f.is_file()]
                files.sort(key=lambda x: x[1])  # Sort by modification time
                
                for file_path, _ in files:
                    if total_size <= max_size_bytes:
                        break
                    file_size = file_path.stat().st_size
                    file_path.unlink()
                    total_size -= file_size
        
        @staticmethod
        def enforce_log_retention(logs_dir: Path, max_files: int = 5) -> None:
            """Enforce log file retention.
            
            Args:
                logs_dir: Logs directory
                max_files: Maximum number of log files to retain
            """
            log_files = sorted(
                [f for f in logs_dir.glob('*.log') if f.is_file()],
                key=lambda f: f.stat().st_mtime,
                reverse=True
            )
            
            # Keep only the most recent files
            for old_file in log_files[max_files:]:
                old_file.unlink()
    ```

    ```python
    # SRC/cuepoint/utils/privacy.py
    from pathlib import Path
    import shutil
    
    class DataDeletionManager:
        """Manage user data deletion."""
        
        @staticmethod
        def delete_user_data(data_type: str, user_confirm: bool = True) -> bool:
            """Delete user data.
            
            Args:
                data_type: Type of data to delete ("cache", "logs", "config", "all")
                user_confirm: Require user confirmation
            
            Returns:
                True if deletion successful
            """
            from cuepoint.utils.paths import AppPaths
            
            if data_type == "cache":
                cache_dir = AppPaths.cache_dir()
                if cache_dir.exists():
                    shutil.rmtree(cache_dir)
                    cache_dir.mkdir(parents=True, exist_ok=True)
                return True
            
            elif data_type == "logs":
                logs_dir = AppPaths.logs_dir()
                if logs_dir.exists():
                    for log_file in logs_dir.glob('*.log'):
                        log_file.unlink()
                return True
            
            elif data_type == "config":
                # Warning: This will reset all settings
                config_dir = AppPaths.config_dir()
                if config_dir.exists():
                    shutil.rmtree(config_dir)
                    config_dir.mkdir(parents=True, exist_ok=True)
                return True
            
            elif data_type == "all":
                # Delete all user data (with confirmation)
                if user_confirm:
                    # Show confirmation dialog
                    pass
                
                DataDeletionManager.delete_user_data("cache", user_confirm=False)
                DataDeletionManager.delete_user_data("logs", user_confirm=False)
                DataDeletionManager.delete_user_data("config", user_confirm=False)
                return True
            
            return False
    ```
